32 enum PeerWireStates {
33  EARLY_ABORTING, //When we were instructed to close the connection before its final establishment
34  ACTIVE_ABORTING, //When we are closing the connection
35  PASSIVE_ABORTING, //When the peer has closed the connection
36  INITIAL, //When the thread has just initialized
37  CONNECTED, //When the connection has been established, app layer acceptance pending ,,,
38  ACTIVE_HANDSHAKE, //When initating the handshake procedure
39  PASSIVE_HANDSHAKE, //When the remote peer initiates the handshake procedure (not actually required, just for
40  //state transition clearness).
41  SENT_HANDSHAKE, //Have ONLY sent a handshake msg
42  RECEIVED_HANDSHAKE, //Have ONLY received a handshake msg
43  HANDSHAKE_COMPLETE, //Have completed the handshake procedure
44  //BITFIELD_PENDING, //Have not sent the bitfield because it is empty
45  BITFIELD_COMPLETE, //Have sent the bitfield
46  ANTI_SNUBBING
47 //In anti-snubbing state i.e. not uploading unless as an optimistic unchoke
48 };
